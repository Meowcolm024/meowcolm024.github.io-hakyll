(define (naive-person fname lname)
    (list 'person fname lname))
(define (person fn ln)
    (let ((fname fn)
          (lname ln))
         (define (dispatch message)
            (cond
                ((eq? message 'first-name) fname)
                ((eq? message 'last-name) lname)
                ((eq? message 'set-fname)
                    (lambda (new-name) (set! fname new-name)))
                ((eq? message 'copy) (person fname lname))
                ((eq? message 'show) 
                    (string-append fname " " lname))
                (else (error "Unknown message"))))
        dispatch))
;; we convert message-passing style to function call
(define (first-name p) (p 'first-name))
(define (last-name p) (p 'last-name))
(define (show p) (p 'show))
(define (copy p) (p 'copy))
(define (set-fname! p new-name) ((p 'set-fname) new-name))
(define (person+ bs ag)
    (let ((base bs) ;; base class object
          (age ag))
         (define (dispatch message)
            (cond
                ((eq? message 'age) age)
                ((eq? message 'grow) (set! age (+ age 1)))
                ((eq? message 'copy) (person+ (base 'copy) age))
                ((eq? message 'show)    ;; override
                    (string-append (base message) " " (number->string age)))
                (else (base message)))) ;; inherit base class methods
    dispatch))

(define (age p) (p 'age))
(define (grow! p) (p 'grow))
(define (person+1 fn ln ag)
    (person+ (person fn ln) ag))
;; instances
(define alice (naive-person "kat" "alice"))
(define jack (person "mat" "jack"))
(define old-jack (person+ jack 60))
(define ben (person+1 "han" "ben" 10))                      
(define ben-2 (copy ben))
